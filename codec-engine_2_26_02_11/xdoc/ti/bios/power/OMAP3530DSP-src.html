    37    package ti.bios.power;
    38    
    39    
    40    /*! DSP configuration module for the OMAP3530 device.
    41     *
    42     *  Configuration parameters for the Local Power Manager running
    43     *  on the DSP core of an OMAP3530 device.
    44     */
    45    @Template("./config.xdt")
    46    
    47    metaonly module OMAP3530DSP inherits ITMS320CDM644x {
    48    
    49        //! dsp runtime context buffer
    50        struct RuntimeContext {
    51            UInt32  len;            //! length of context buffer
    52            Bool    useUserFxns;    //! override save/restore functions
    53            String  saveFxn;        //! context save function name
    54            String  restoreFxn;     //! context restore function name
    55        };
    56    
    57        //! maximum allowable size for context buffer
    58        readonly config unsigned CONTEXT_BUFFER_MAX_SIZE = 0x40000;
    59    
    60        //! default value for dsp runtime context buffer
    61        config RuntimeContext dspRC = {
    62            len:            0x40000,
    63            useUserFxns:    false,
    64            saveFxn:        "",
    65            restoreFxn:     "",
    66        };
    67    
    68        //! default edma channel to use for save/restore transfers
    69        config UInt32 edmaChannel = 45;
    70    
    71        /*! name given to target memory range for storing context data
    72         *
    73         *  This must be set to a valid target memory range defined in the
    74         *  linker command file's memory section.
    75         */
    76        config string targetMemoryRangeName = "";
    77    
    78        //! cpu name, useful for adding validation checks
    79        readonly config string CPU_NAME = "DSP";
    80    
    81        //! stop BIOS clock when hibernating
    82        config Bool stopBiosClockOnHibernate;
    83    
    84        //! stop CPU load monitoring when hibernating
    85        config Bool stopCpuLoadMonitorOnHibernate = true;
    86    
    87        //! WUGEN wake event mask
    88        config UInt32 wugenEventMask[3];
    89    }
