    34    /*!
    35     *  ======== Settings ========
    36     *  Global OS Abstract Layer Configuration
    37     */
    38    @Template("./Settings.xdt")
    39    
    40    metaonly module Settings inherits ti.sdo.ce.osal.IOsal {
    41        /*!
    42         *  ======== ipcKeyBase ========
    43         *  Default value for IPC key needed by linux SemMP module for implementing
    44         *  a critical section.
    45         *  The default value of the ipcKeyBase is ascii code for "LASO".
    46         *
    47         *  WARNING: This value should only be changed if it conflicts with
    48         *  another IPC key in the system that cannot be changed. If this value
    49         *  is changed, all programs using Codec Engine that will be run
    50         *  simultaneously must have the ipcKeyBase configured to the new value.
    51         *
    52         * @_nodoc
    53         */
    54        config UInt32 ipcKeyBase = 0x4C41534F;
    55    
    56        /*!
    57         *  ======== maxCbListSize ========
    58         *  Maximum length of Contiguous Block List.
    59         *
    60         *  The maximum length of the contiguous block list
    61         *  used to speed up virtual-to-physical address translation of Arm
    62         *  buffer addresses.  The length indicates how many contiguous blocks
    63         *  are kept in the list, and was introduced to prevent performance
    64         *  degradation for applictions whose pattern of memory accesses made
    65         *  this list grow very large.  Most "regular" applications will not see
    66         *  a difference in performance with the introduction of this cut-off.
    67         *
    68         *  Applies to Linux only.
    69         */
    70        config UInt32 maxCbListSize = 100;
    71    
    72        /*!
    73         *  ======== debug ========
    74         *  Enable the debug profile of the osal Linux module.
    75         *
    76         *  This parameter is no longer supported.  See the profile config param
    77         *  in the ti.sdo.ce.global.Settings module.
    78         *
    79         *  @_nodoc
    80         */
    81        config Bool debug = false;
    82    }
