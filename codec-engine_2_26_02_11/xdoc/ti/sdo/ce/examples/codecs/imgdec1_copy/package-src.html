    35    /*!
    36     *  ======== package.xdc ========
    37     *  Simple Codec Engine compatible image 1.0 "decoder", wrapping the image 1.0
    38     *  decode copy codec distributed with xDAIS.
    39     *
    40     *  This codec illustrates the simplest integration into the Codec
    41     *  Engine.  Only three files are required to integrate into the codec engine:
    42     *  @p(dlist)
    43     *      - `package.xdc`
    44     *          This file is required to declare a name for this package and to
    45     *          list all codec interfaces available.
    46     *      - `IMGDEC1_COPY.xdc`
    47     *          This file specifies a codec (named IMGDEC1_COPY) that is
    48     *          provided by this package.
    49     *      - `IMGDEC1_COPY.xs`
    50     *          This file implements the functions required by the ICodec
    51     *          interface; these functions enable the codec to describe its
    52     *          non-functional requirements such as stack size.
    53     *
    54     *  Integration into the Codec Engine requires no change to the codecs
    55     *  libraries.
    56     */

    58        module IMGDEC1_COPY;
    59    }
