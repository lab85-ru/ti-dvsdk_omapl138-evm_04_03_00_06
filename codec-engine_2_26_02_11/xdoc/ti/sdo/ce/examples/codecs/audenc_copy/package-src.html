    33    /*!
    34     *  ======== package.xdc ========
    35     *  Codec Engine compatible audio encoder.
    36     *
    37     *  This package provides a reference IAUDENC-compliant algorithm, and the
    38     *  necessary support required for integration into the Codec Engine.
    39     *
    40     *  Only three files are required to integrate into the Codec Engine:
    41     *  @p(dlist)
    42     *      - `package.xdc`
    43     *          This file is required to declare a name for this package and to
    44     *          list all codec interfaces available.
    45     *      - `AUDENC_COPY.xdc`
    46     *          This file specifies a codec (named AUDENC_COPY) that is
    47     *          provided by this package.
    48     *      - `AUDENC_COPY.xs`
    49     *          This file implements the functions required by the ICodec
    50     *          interface; these functions enable the codec to describe its
    51     *          non-functional requirements such as stack size.
    52     *  @p
    53     *
    54     *  All other files in this package are simply the implementation of the codec
    55     *  as specified by xDM's IAUDENC interface.  Integration into the Codec Engine
    56     *  requires no change to the codecs sources.
    57     *
    58     *  As the IAUDENC interface is deprecated, the xDAIS product doesn't provide
    59     *  an example of an algorithm that implements the IAUDENC interface.  This
    60     *  package serves two purposes:
    61     *  @p(blist)
    62     *      - Simple reference implementation of an algorithm that implements the
    63     *        IAUDENC interface
    64     *      - Example of a package that both provides an algorithm's xDAIS/xDM
    65     *        libraries and implements the ti.sdo.ce.ICodec interface.
    66      *  @p
    67     */
    68    package ti.sdo.ce.examples.codecs.audenc_copy [1, 0, 0] {
    69        module AUDENC_COPY;
    70    }
