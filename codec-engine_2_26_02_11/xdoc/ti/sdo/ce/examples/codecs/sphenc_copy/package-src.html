    33    /*!
    34     *  ======== package.xdc ========
    35     *  Simple Codec Engine compatible speech "encoder".
    36     *
    37     *  This codec illustrates the simplest integration into the Codec
    38     *  Engine.  Only three files are required to integrate into the Codec Engine:
    39     *  @p(dlist)
    40     *      - `package.xdc`
    41     *          This file is required to declare a name for this package and to
    42     *          list all codec interfaces available.
    43     *      - `SPHENC_COPY.xdc`
    44     *          This file specifies a codec (named SPHENC_COPY) that is
    45     *          provided by this package.
    46     *      - `SPHENC_COPY.xs`
    47     *          this file implements the functions required by the ICodec
    48     *          interface; these functions enable the codec to describe its
    49     *          non-functional requirements such as stack size.
    50     *
    51     *  All other files in this package are simply the implementation of the codec
    52     *  as specified by xDM.  Thus, integration into the Codec Engine requires
    53     *  no change to the codecs sources.
    54     */

    56        module SPHENC_COPY;
    57    }
